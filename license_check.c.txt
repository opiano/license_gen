#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <ctype.h>
#include <openssl/sha.h>
#include <openssl/pem.h>
#include <openssl/evp.h>
#include <openssl/bio.h>

// ---------- HW ID ----------

void hex_print(unsigned char *hash, char *output) {
    for (int i = 0; i < SHA256_DIGEST_LENGTH; i++)
        sprintf(output + (i * 2), "%02x", hash[i]);
}

int read_serial(char *out, size_t len) {
    FILE *fp = fopen("/proc/device-tree/hat/custom_1", "r");
    if (!fp) return -1;

    fgets(out, len, fp);
    fclose(fp);
    out[strcspn(out, "\n")] = 0;
    return 0;
}

int read_mac(char *out, size_t len) {
    FILE *fp = fopen("/sys/class/net/eth0/address", "r");
    if (!fp) return -1;

    if (!fgets(out, len, fp)) {
        fclose(fp);
        return -1;
    }

    fclose(fp);

    out[strcspn(out, "\n")] = 0;

    for (int i = 0; out[i]; i++) {
        out[i] = tolower((unsigned char)out[i]);
    }

    return 0;
}

void generate_hw_id(char *hw_id_out) {
    char serial[128] = {0};
    char mac_str[64] = {0};
    char combined[256];
    int i, j;

    if (read_serial(serial, sizeof(serial)) != 0) {
        printf("Failed to read serial\n");
        strcpy(serial, "UNKNOWN");
    }
    //strcpy(serial, "123456");

    if (read_mac(mac_str, sizeof(mac_str)) != 0) {
        printf("Failed to read MAC\n");
        strcpy(mac_str, "UNKNOWN");
    }

    i = 0, j = 0;
    while (mac_str[i]) {
        if (mac_str[i] != ':') {
            mac_str[j++] = mac_str[i];
        }
        i++;
    }
    mac_str[j] = '\0';

    //strcpy(mac_str, "0000f0123456");

    snprintf(combined, sizeof(combined), "%s%s", serial, mac_str);

    unsigned char hash[SHA256_DIGEST_LENGTH];
    SHA256((unsigned char*)combined, strlen(combined), hash);

    hex_print(hash, hw_id_out);
}

// ---------- Base64 decode ----------

unsigned char* base64_decode(const char *input, int *out_len) {
    BIO *bio, *b64;
    int len = strlen(input);

    unsigned char *buffer = malloc(len);
    if (!buffer) return NULL;

    memset(buffer, 0, len);

    bio = BIO_new_mem_buf((void*)input, -1);
    b64 = BIO_new(BIO_f_base64());
    bio = BIO_push(b64, bio);

    BIO_set_flags(bio, BIO_FLAGS_BASE64_NO_NL);
    *out_len = BIO_read(bio, buffer, len);

    BIO_free_all(bio);
    return buffer;
}

// ---------- 서명 검증 ----------

int verify_signature(const char *data, const char *signature_b64, const char *pubkey_path) {
    FILE *fp = fopen(pubkey_path, "r");
    if (!fp) {
        printf("Failed to open public key\n");
        return -1;
    }

    EVP_PKEY *pubkey = PEM_read_PUBKEY(fp, NULL, NULL, NULL);
    fclose(fp);
    if (!pubkey) {
        printf("Failed to read public key\n");
        return -1;
    }

    EVP_MD_CTX *ctx = EVP_MD_CTX_new();
    EVP_PKEY_CTX *pctx;

    if (!EVP_DigestVerifyInit(ctx, &pctx, EVP_sha256(), NULL, pubkey)) {
        EVP_MD_CTX_free(ctx);
        EVP_PKEY_free(pubkey);
        return -1;
    }

    if (!EVP_DigestVerifyUpdate(ctx, data, strlen(data))) {
        EVP_MD_CTX_free(ctx);
        EVP_PKEY_free(pubkey);
        return -1;
    }

    int sig_len = 0;
    unsigned char *sig = base64_decode(signature_b64, &sig_len);
    if (!sig) {
        EVP_MD_CTX_free(ctx);
        EVP_PKEY_free(pubkey);
        return -1;
    }

    int ret = EVP_DigestVerifyFinal(ctx, sig, sig_len);

    free(sig);
    EVP_MD_CTX_free(ctx);
    EVP_PKEY_free(pubkey);

    return ret == 1 ? 0 : -1;
}

// ---------- license.txt 파싱 ----------

int parse_license_txt(const char *path,
                      char *hw_id,
                      char *expire,
                      char *signature)
{
    FILE *fp = fopen(path, "r");
    if (!fp) return -1;

    char line[1024];

    while (fgets(line, sizeof(line), fp)) {
        char key[64] = {0};
        char value[512] = {0};

        line[strcspn(line, "\r\n")] = 0;

        if (sscanf(line, "%63[^=]=\"%511[^\"]\"", key, value) == 2) {

            if (strcmp(key, "hw_id") == 0) {
                strncpy(hw_id, value, 64);
            }
            else if (strcmp(key, "expire") == 0) {
                strncpy(expire, value, 31);
            }
            else if (strcmp(key, "signature") == 0) {
                strncpy(signature, value, 511);
            }
        }
    }

    fclose(fp);

    if (strlen(hw_id) == 0 || strlen(expire) == 0 || strlen(signature) == 0) {
        return -1;
    }

    return 0;
}

// ---------- MAIN ----------

int main() {
    char hw_id[65] = {0};
    char license_hw_id[65] = {0};
    char expire[32] = {0};
    char signature[512] = {0};

    // 1. HW ID 생성
    generate_hw_id(hw_id);
    printf("HW_ID: %s\n", hw_id);

    // 2. license.txt 읽기
    if (parse_license_txt("./license.txt",
                          license_hw_id,
                          expire,
                          signature) != 0)
    {
        printf("Failed to read license file\n");
        return -1;
    }

    // 3. HW ID 비교
    if (strcmp(hw_id, license_hw_id) != 0) {
        printf("HW ID mismatch\n");
        return -1;
    }

    // 4. 서명 데이터 생성
    char data[256];
    snprintf(data, sizeof(data), "%s|%s", license_hw_id, expire);

    // 5. 서명 검증
    if (verify_signature(data, signature, "./scon_public_key.pem") != 0) {
        printf("Signature verification failed\n");
        return -1;
    }

    printf("License OK\n");
    printf("Expire: %s\n", expire);

    return 0;
}
